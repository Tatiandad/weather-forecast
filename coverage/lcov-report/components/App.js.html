  processData = () => {
    const days = new Map();
    this.state.list.forEach((item) => {
      const day = this.getDate(item.dt_txt);
      if( !days[day] ) days[day] = []

  getDate = (dateString) => {
    const regex = /([0-9]{4}-[0-9]{2}-[0-9]{2})/;
